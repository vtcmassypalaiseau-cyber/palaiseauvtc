        // Calculate route and price

        function calculateRoute() {

            const departure = document.getElementById('departure').value;

            const destination = document.getElementById('destination').value;


            if (!departure || !destination) return;


            const request = {

                origin: departure,

                destination: destination,

                travelMode: 'DRIVING'

            };


            directionsService.route(request, (result, status) => {

                if (status === 'OK') {

                    directionsRenderer.setDirections(result);

                    

                    const route = result.routes[0];

                    const leg = route.legs[0];

                    

                    distanceKm = leg.distance.value / 1000;

                    durationMinutes = leg.duration.value / 60;

                    

                    document.getElementById('distance').textContent = distanceKm.toFixed(1);

                    document.getElementById('duration-est').textContent = Math.round(durationMinutes) + ' min';

                    document.getElementById('routeInfo').style.display = 'block';

                    

                    calculatePrice();

                }

            });

        }


        // Calculate price with all surcharges

        function calculatePrice() {

            const vehicle = vehicles[selectedVehicle];

            const date = new Date(document.getElementById('departureDate').value + 'T' + document.getElementById('departureTime').value);

            

            let basePrice = 0;

            let surcharges = 0;


            if (selectedServiceType === 'trajet') {

                basePrice = (distanceKm * vehicle.pricePerKm) + (durationMinutes * vehicle.pricePerMin);

            } else {

                const hours = parseFloat(document.getElementById('duration').value) || 1;

                basePrice = hours * vehicle.hourlyRate;

            }


            basePrice *= vehicle.multiplier;


            // Hour surcharge

            const hours = date.getHours();

            if (hours >= 6 && hours < 9) {

                surcharges += basePrice * 0.20; // Rush hour

                document.getElementById('surchargeLineHour').style.display = 'flex';

                document.getElementById('surchargeHourDisplay').textContent = '+' + (basePrice * 0.20).toFixed(2) + '€';

            } else if (hours >= 22 || hours < 6) {

                surcharges += basePrice * 0.20; // Night

                document.getElementById('surchargeLineHour').style.display = 'flex';

                document.getElementById('surchargeHourDisplay').textContent = '+' + (basePrice * 0.20).toFixed(2) + '€';

            } else {

                document.getElementById('surchargeLineHour').style.display = 'none';

            }


            // Day surcharge

            const day = date.getDay();

            if (day === 0 || day === 6) {

                surcharges += basePrice * 0.10; // Weekend

                document.getElementById('surchargeLineDay').style.display = 'flex';

                document.getElementById('surgechargeDayDisplay').textContent = '+' + (basePrice * 0.10).toFixed(2) + '€';

            } else {

                document.getElementById('surchargeLineDay').style.display = 'none';

            }


            let totalPrice = basePrice + surcharges;

            let discount = 0;


            // Apply promo if any

            const promoCode = document.getElementById('promoCode').value;

            if (promoCode && promoCodes[promoCode.toUpperCase()]) {

                const promo = promoCodes[promoCode.toUpperCase()];

                if (promo.type === 'percent') {

                    discount = totalPrice * (promo.discount / 100);

                } else {

                    discount = promo.discount;

                }

                document.getElementById('promoPriceDisplay').style.display = 'flex';

                document.getElementById('promoAmount').textContent = '-' + discount.toFixed(2) + '€';

            } else {

                document.getElementById('promoPriceDisplay').style.display = 'none';

            }


            totalPrice -= discount;


            document.getElementById('basePriceDisplay').textContent = basePrice.toFixed(2) + '€';

            document.getElementById('totalPrice').textContent = totalPrice.toFixed(2) + '€';

        }

        }
